#4: Functions

However, in ECMAScript 4 and lower (everything up to IE8), undefined is a mutable object:


  window.undefined = "Dimitri Mestdagh";
  
  var hasName = function(person) {
    return person.name !== undefined;
  };
  
  console.log(hasName({
    name: "Dimitri Mestdagh" // This will return false in IE8!!!
  }));
